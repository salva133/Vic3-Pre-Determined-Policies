# Prescribes the ideological path of every non-player country according to the
# pdp_predetermined_policies game rule. Called once from on_game_started_after_lobby.
#
# The chose_* variables are interest-group scoped in vanilla: they are set on the one
# interest group that has committed to a path. On that IG the base game cuts the leader
# weight of every off-path ideology — moderates included — to a tenth, so the group keeps
# putting forward leaders of the chosen path. This effect seats each marker on an interest
# group nearly every country has: the Trade Unions for the left-wing paths, exactly as
# vanilla does, and the Armed Forces for the right-wing paths, a stand-in for the governing
# IG vanilla would otherwise pick out dynamically. The ?= guard skips any country lacking the IG.
pdp_apply_predetermined_policies = {
	if = {
		limit = { has_game_rule = pdp_vanguardism }
		every_country = {
			limit = { is_player = no }
			ig:ig_trade_unions ?= { set_variable = chose_vanguardism }
		}
	}
	else_if = {
		limit = { has_game_rule = pdp_communism }
		every_country = {
			limit = { is_player = no }
			ig:ig_trade_unions ?= { set_variable = chose_communism }
		}
	}
	else_if = {
		limit = { has_game_rule = pdp_anarchism }
		every_country = {
			limit = { is_player = no }
			ig:ig_trade_unions ?= { set_variable = chose_anarchism }
		}
	}
	else_if = {
		limit = { has_game_rule = pdp_fascism }
		every_country = {
			limit = { is_player = no }
			ig:ig_armed_forces ?= { set_variable = chose_fascism }
		}
	}
	else_if = {
		limit = { has_game_rule = pdp_ethno_nationalism }
		every_country = {
			limit = { is_player = no }
			ig:ig_armed_forces ?= { set_variable = chose_ethno_nationalism }
		}
	}
	else_if = {
		limit = { has_game_rule = pdp_conservatism }
		every_country = {
			limit = { is_player = no }
			ig:ig_armed_forces ?= { set_variable = chose_conservatism }
		}
	}
	else_if = {
		limit = { has_game_rule = pdp_historical }
		pdp_apply_historical_policies = yes
	}
	else_if = {
		limit = { has_game_rule = pdp_absolute_insanity }
		pdp_apply_absolute_insanity = yes
	}
}

# Seats a chose_* marker on a country's Trade Unions, the interest group vanilla uses
# for the left-wing paths. PATH = the chose_* marker to set. Players are never touched;
# a country lacking the interest group is skipped by the ?= guard.
pdp_seat_left_path = {
	if = {
		limit = { is_player = no }
		ig:ig_trade_unions ?= { set_variable = $PATH$ }
	}
}

# Seats a chose_* marker on a country's Armed Forces, the stand-in this mod uses for the
# right-wing paths. PATH = the chose_* marker to set. Players are never touched;
# a country lacking the interest group is skipped by the ?= guard.
pdp_seat_right_path = {
	if = {
		limit = { is_player = no }
		ig:ig_armed_forces ?= { set_variable = $PATH$ }
	}
}

# The pdp_historical option: seats each country on the ideological path its historical
# government had settled into by 1936. The regimes are keyed on the 1836 starting tags,
# because the 1936 successor states do not exist at game start (Nazi Germany rides on
# Prussia, Fascist Italy on Sardinia-Piedmont). Every country the curated list does not
# name keeps the Conservative status quo of the interwar constitutional order.
pdp_apply_historical_policies = {
	every_country = {
		limit = { is_player = no }
		pdp_seat_right_path = { PATH = chose_conservatism }
	}

	# Fascist and authoritarian-right regimes.
	c:PRU ?= { pdp_seat_right_path = { PATH = chose_fascism } }			# Nazi Germany
	c:SAR ?= { pdp_seat_right_path = { PATH = chose_fascism } }			# Fascist Italy
	c:AUS ?= { pdp_seat_right_path = { PATH = chose_fascism } }			# Austrofascism
	c:POR ?= { pdp_seat_right_path = { PATH = chose_fascism } }			# Estado Novo
	c:SPA ?= { pdp_seat_right_path = { PATH = chose_fascism } }			# Nationalist Spain
	c:JAP ?= { pdp_seat_right_path = { PATH = chose_fascism } }			# Imperial Japan

	# Kemalist Turkey: a one-party nationalist republic.
	c:TUR ?= { pdp_seat_right_path = { PATH = chose_ethno_nationalism } }

	# The Soviet Union: a Leninist vanguard-party state. The path sits on the Trade Unions,
	# so the baseline Conservatism marker on its Armed Forces is cleared first.
	c:RUS ?= {
		if = {
			limit = { is_player = no }
			ig:ig_armed_forces ?= { remove_variable = chose_conservatism }
			pdp_seat_left_path = { PATH = chose_vanguardism }
		}
	}
}

# The pdp_absolute_insanity option: draws one of the six paths at random, independently
# for every AI country. random_list uses the synchronized game RNG, so the outcome is
# deterministic per seed and safe in multiplayer.
pdp_apply_absolute_insanity = {
	every_country = {
		limit = { is_player = no }
		random_list = {
			1 = { pdp_seat_left_path = { PATH = chose_vanguardism } }
			1 = { pdp_seat_left_path = { PATH = chose_communism } }
			1 = { pdp_seat_left_path = { PATH = chose_anarchism } }
			1 = { pdp_seat_right_path = { PATH = chose_fascism } }
			1 = { pdp_seat_right_path = { PATH = chose_ethno_nationalism } }
			1 = { pdp_seat_right_path = { PATH = chose_conservatism } }
		}
	}
}
